-- neon pulse

	while container.Parent do
		TweenService:Create(stroke, TweenInfo.new(0.4), {Transparency = 0.05}):Play()
		task.wait(0.4)
		TweenService:Create(stroke, TweenInfo.new(0.4), {Transparency = 0.35}):Play()

	end
end)

-- auto close
task.delay(5, function()
	TweenService:Create(container, TweenInfo.new(0.4), {BackgroundTransparency = 1}):Play()
	for _, v in ipairs(container:GetChildren()) do
		if v:IsA("TextLabel") then
			TweenService:Create(v, TweenInfo.new(0.4), {TextTransparency = 1}):Play()
		end
	end
	TweenService:Create(stroke, TweenInfo.new(0.4), {Transparency = 1}):Play()
	TweenService:Create(blur, TweenInfo.new(0.4), {Size = 0}):Play()
	task.wait(0.45)
	gui:Destroy()
	blur:Destroy()
end)
